```ts
'use server'
import { z } from 'zod'
import { revalidatePath } from 'next/cache'
import { requireAdmin } from '@/lib/auth/require-admin'
import { createAdminClient } from '@/lib/supabase/admin'

const ArchiveSeasonSchema = z.object({ season: z.number().int() })

export async function archiveSeason(formData: FormData) {
  await requireAdmin()
  const parsed = ArchiveSeasonSchema.safeParse({ season: Number(formData.get('season')) })
  if (!parsed.success) return { error: 'Invalid input' }
  const admin = createAdminClient()
  // Idempotent — only set ended_at if null
  const { error } = await admin
    .from('seasons')
    .update({ ended_at: new Date().toISOString() })
    .eq('season', parsed.data.season)
    .is('ended_at', null)
  if (error) return { error: error.message }
  revalidatePath('/admin/season-rollover')
  revalidatePath('/admin')
  revalidatePath('/')
  revalidatePath('/standings')
  return { ok: true }
}
```
