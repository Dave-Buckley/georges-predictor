```ts
// src/lib/profile/stats.ts
export interface SeasonStats {
  season: number
  totalPoints: number
  rank: number | null
  predictionAccuracy: number        // correctResults / totalPredictions (0..1)
  correctResults: number
  correctScores: number
  bonusConfirmationRate: number     // confirmed / (confirmed + pending + rejected)
  losStatus: 'active' | 'eliminated' | 'winner' | 'not-participating'
  losTeamsUsed: number
  losWins: number
  gwWinnerCount: number
  achievements: Achievement[]
}

export interface Achievement {
  kind: 'gw-winner' | 'los-winner' | 'h2h-survivor' | 'pre-season-all-correct' | 'pre-season-category-correct'
  label: string
  detail?: string
}

export function aggregateSeasonStats(input: {
  predictionScores: PredictionScoreRow[]
  bonusAwards: BonusAwardRow[]
  prizeAwards: PrizeAwardRow[]
  preSeasonAward: PreSeasonAwardRow | null
  losPicks: LosPickRow[]
  losCompetitions: LosCompetitionRow[]
  h2hSteals: H2hStealRow[]
  gameweeks: GameweekRow[]
  allMemberTotals: { memberId: string; totalPoints: number }[]  // for rank derivation
  memberId: string
  season: number
}): SeasonStats
```
Zero imports. Pure. Throws on memberId not in allMemberTotals.
