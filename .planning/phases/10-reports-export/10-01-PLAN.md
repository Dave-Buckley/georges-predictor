3. Create `src/lib/email/send-attachments.ts`:
   ```typescript
   import { getResend, DEFAULT_FROM } from './client'

   export interface EmailAttachment {
     filename: string
     content: Buffer          // caller passes Buffer; helper converts to base64
   }
   export interface SendWithAttachmentsInput {
     to: string | string[]
     subject: string
     html: string
     attachments?: EmailAttachment[]
     from?: string
   }

   export async function sendWithAttachments(input: SendWithAttachmentsInput): Promise<{ id?: string; error?: string }> {
     const resend = getResend()
     if (!resend) {
       console.warn('[sendWithAttachments] RESEND_API_KEY not configured — skipping')
       return { error: 'RESEND_API_KEY not configured' }
     }
     const { data, error } = await resend.emails.send({
       from: input.from ?? DEFAULT_FROM,
       to: input.to,
       subject: input.subject,
       html: input.html,
       attachments: input.attachments?.map(a => ({
         filename: a.filename,
         content: a.content.toString('base64'),
       })),
     })
     if (error) {
       const msg = typeof error === 'string' ? error : (error.message ?? 'Failed to send email')
       console.error('[sendWithAttachments] Resend error:', error)
       return { error: msg }
     }
     return { id: data?.id }
   }
   ```
