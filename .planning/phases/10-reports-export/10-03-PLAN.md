   export async function sleep(ms: number) { return new Promise(r => setTimeout(r, ms)) }

   export interface SendSummary { /* see <interfaces> */ }

   async function checkAdminGate(key: 'email_weekly_personal_enabled' | 'email_weekly_group_enabled'): Promise<boolean> {
     const s = createAdminClient()
     const { data } = await s.from('admin_settings').select(key).single()
     return Boolean(data?.[key])
   }

   export async function sendPersonalReports(gwId: string): Promise<SendSummary> {
     const summary: SendSummary = { gwId, reportType: 'personal', sent: 0, failed: 0, skipped: 0, skippedByAdmin: false }
     if (!await checkAdminGate('email_weekly_personal_enabled')) { summary.skippedByAdmin = true; return summary }

     const supabase = createAdminClient()
     const data = await gatherGameweekData(gwId)
     const { data: members } = await supabase.from('members').select('id, display_name, email, email_weekly_personal, user_id').not('user_id','is',null).eq('email_weekly_personal', true)
     const { data: alreadyLogged } = await supabase.from('member_report_log').select('member_id').eq('gameweek_id', gwId).eq('report_type', 'personal')
     const loggedSet = new Set((alreadyLogged ?? []).map(r => r.member_id))

     for (const m of members ?? []) {
       if (loggedSet.has(m.id)) { summary.skipped++; continue }
       try {
         const pdfBuffer = await renderPersonalWeeklyPdf(data, m.id)
         const rank = data.standings.find(s => s.memberId === m.id)?.rank ?? null
         const weeklyPoints = data.standings.find(s => s.memberId === m.id)?.weeklyPoints ?? 0
         const html = await renderEmail(PersonalWeeklyEmail({
           displayName: m.display_name, gwNumber: data.gwNumber, weeklyPoints, rank,
           gameweekLink: `${process.env.NEXT_PUBLIC_APP_URL}/gameweeks/${data.gwNumber}`,
         }))
         const { error } = await sendWithAttachments({
           to: m.email, subject: `GW${data.gwNumber} — your weekly breakdown`, html,
           attachments: [{ filename: `gw${data.gwNumber}-${slug(m.display_name)}.pdf`, content: pdfBuffer }],
         })
         if (error) throw new Error(error)
         await supabase.from('member_report_log').insert({ member_id: m.id, gameweek_id: gwId, report_type: 'personal' })
         summary.sent++
       } catch (err) {
         await supabase.from('admin_notifications').insert({
           type: 'report_send_failed',
           title: `Personal PDF failed for ${m.display_name}`,
           message: String(err),
         })
         summary.failed++
       }
       await sleep(550)   // 2 req/sec Resend ceiling + 50ms safety margin
     }
     return summary
   }

   // sendGroupReports — one PDF rendered ONCE, N sends (ignore personal flag; use email_weekly_group). Same idempotency + pacing pattern.

3. Implement `src/app/api/reports/send-weekly/route.ts`:
   ```typescript
   import { NextResponse } from 'next/server'
   import { sendGroupReports, sendPersonalReports, sendAdminWeekly } from '@/lib/reports/orchestrate'
   import { createAdminClient } from '@/lib/supabase/admin'

   export const maxDuration = 60
   export const dynamic = 'force-dynamic'

   export async function POST(req: Request) {
     const auth = req.headers.get('authorization')
     if (auth !== `Bearer ${process.env.CRON_SECRET}`) {
       return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
     }
     const { gameweek_id } = await req.json()
     if (!gameweek_id) return NextResponse.json({ error: 'gameweek_id required' }, { status: 400 })

     const [groupSummary, personalSummary, adminSummary] = await Promise.all([
       sendGroupReports(gameweek_id),
       sendPersonalReports(gameweek_id),
       sendAdminWeekly(gameweek_id),
     ])

     // Once all three complete (success or partial), mark gameweeks.reports_sent_at
     const supabase = createAdminClient()
     await supabase.from('gameweeks').update({ reports_sent_at: new Date().toISOString() }).eq('id', gameweek_id).is('reports_sent_at', null)

     return NextResponse.json({ group: groupSummary, personal: personalSummary, admin: adminSummary })
   }

   import 'server-only'
   import { createAdminClient } from '@/lib/supabase/admin'
   import { gatherGameweekData } from './_data/gather-gameweek-data'
   import { renderKickoffBackupPdf } from './kickoff-backup-pdf'
   import { buildKickoffBackupXlsx } from './kickoff-backup-xlsx'
   import { sendWithAttachments } from '@/lib/email/send-attachments'
   import { render as renderEmail } from '@react-email/components'
   import KickoffBackupEmail from '@/emails/kickoff-backup'

   export async function sendKickoffBackupEmail(gwId: string): Promise<void> {
     const data = await gatherGameweekData(gwId)
     const [pdf, xlsx] = await Promise.all([
       renderKickoffBackupPdf(data),
       Promise.resolve(buildKickoffBackupXlsx(data)),
     ])
     const admins = [process.env.ADMIN_EMAIL_GEORGE, process.env.ADMIN_EMAIL_DAVE].filter(Boolean) as string[]
     if (!admins.length) throw new Error('No admin emails configured for kickoff backup')

     const html = await renderEmail(KickoffBackupEmail({
       gwNumber: data.gwNumber,
       memberCount: data.standings.length,
       kickoffIso: data.fixtures[0]?.kickoffIso ?? '',
     }))

     const { error } = await sendWithAttachments({
       to: admins,
       subject: `Backup — GW${data.gwNumber} all predictions as of kickoff`,
       html,
       attachments: [
         { filename: `kickoff-backup-gw${data.gwNumber}.pdf`, content: pdf },
         { filename: `kickoff-backup-gw${data.gwNumber}.xlsx`, content: xlsx },
       ],
     })
     if (error) throw new Error(error)
   }

   export async function maybeSendKickoffBackup(supabase = createAdminClient()): Promise<void> {
     const { data: gws } = await supabase
       .from('gameweeks')
       .select('id, number, kickoff_backup_sent_at')
       .is('kickoff_backup_sent_at', null)
       .order('number')
     if (!gws?.length) return

     for (const gw of gws) {
       const { data: hasKickedOff } = await supabase
         .from('fixtures')
         .select('id')
         .eq('gameweek_id', gw.id)
         .neq('status', 'SCHEDULED')
         .limit(1)
       if (!hasKickedOff?.length) continue

       try {
         await sendKickoffBackupEmail(gw.id)
         await supabase.from('gameweeks')
           .update({ kickoff_backup_sent_at: new Date().toISOString() })
           .eq('id', gw.id)
       } catch (err) {
         await supabase.from('admin_notifications').insert({
           type: 'kickoff_backup_failed',
           title: `Kickoff backup failed for GW${gw.number}`,
           message: String(err),
         })
         // leave flag NULL — next sync retries
       }
     }
   }
   ```
