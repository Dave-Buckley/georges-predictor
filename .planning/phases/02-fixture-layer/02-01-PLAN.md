    Server-side lockout check utility that Phase 3 prediction server actions will call:

    ```typescript
    import { createServerSupabaseClient } from '@/lib/supabase/server'

    export async function canSubmitPrediction(fixtureId: string): Promise<{
      canSubmit: boolean
      reason?: string
      fixture?: { kickoff_time: string; status: string }
    }> {
      const supabase = await createServerSupabaseClient()

      const { data: fixture, error } = await supabase
        .from('fixtures')
        .select('kickoff_time, status')
        .eq('id', fixtureId)
        .single()

      if (error || !fixture) {
        return { canSubmit: false, reason: 'Fixture not found' }
      }

      if (new Date() >= new Date(fixture.kickoff_time)) {
        return { canSubmit: false, reason: 'This fixture has kicked off — predictions are locked', fixture }
      }

      const nonPredictableStatuses = ['IN_PLAY', 'PAUSED', 'FINISHED', 'POSTPONED', 'SUSPENDED', 'CANCELLED', 'AWARDED']
      if (nonPredictableStatuses.includes(fixture.status)) {
        return { canSubmit: false, reason: `Fixture status is ${fixture.status} — predictions are locked`, fixture }
      }

      return { canSubmit: true, fixture }
    }
    ```
