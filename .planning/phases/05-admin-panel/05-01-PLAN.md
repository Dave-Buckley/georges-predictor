   - id (uuid PK default gen_random_uuid()), gameweek_id (uuid NOT NULL REFERENCES gameweeks ON DELETE CASCADE, UNIQUE), bonus_type_id (uuid NOT NULL REFERENCES bonus_types), confirmed (boolean NOT NULL DEFAULT false), confirmed_at (timestamptz), confirmed_by (uuid REFERENCES auth.users), created_at (timestamptz NOT NULL DEFAULT now())

3. `bonus_awards` — per-member bonus award tracking
   - id (uuid PK), gameweek_id (uuid NOT NULL REFERENCES gameweeks ON DELETE CASCADE), member_id (uuid NOT NULL REFERENCES members ON DELETE CASCADE), bonus_type_id (uuid NOT NULL REFERENCES bonus_types), fixture_id (uuid REFERENCES fixtures), awarded (boolean — NULL=pending, true=confirmed, false=rejected), confirmed_by (uuid REFERENCES auth.users), confirmed_at (timestamptz), created_at (timestamptz NOT NULL DEFAULT now()), UNIQUE(gameweek_id, member_id)

   - id (uuid PK), name (text NOT NULL), emoji (text), description (text NOT NULL), trigger_type (text NOT NULL CHECK IN ('auto', 'date', 'manual')), trigger_config (jsonb — stores trigger condition metadata), points_value (int NOT NULL DEFAULT 0), cash_value (int NOT NULL DEFAULT 0 — stored in pence, £10=1000), is_custom (boolean NOT NULL DEFAULT false), created_at (timestamptz NOT NULL DEFAULT now())
