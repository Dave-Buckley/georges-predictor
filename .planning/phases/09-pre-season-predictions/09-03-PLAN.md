export async function calculatePreSeasonAwards(formData: FormData) {
  const auth = await requireAdmin()
  if ('error' in auth) return auth
  const season = Number(formData.get('season'))
  if (!Number.isInteger(season)) return { error: 'Invalid season' }

  const admin = createAdminClient()

  // Fetch season row; gate on actuals_locked_at
  const { data: seasonRow } = await admin.from('seasons').select('*').eq('season', season).single()
  if (!seasonRow || !seasonRow.actuals_locked_at) return { error: 'Season-end actuals are not locked yet' }

  // Fetch all picks for this season with member info
  const { data: picksRows } = await admin.from('pre_season_picks')
    .select('member_id, season, top4, tenth_place, relegated, promoted, promoted_playoff_winner, members(display_name)')
    .eq('season', season)

  const actuals = {
    final_top4: seasonRow.final_top4, final_tenth: seasonRow.final_tenth ?? '',
    final_relegated: seasonRow.final_relegated, final_promoted: seasonRow.final_promoted,
    final_playoff_winner: seasonRow.final_playoff_winner ?? '',
  }

  let flagsAllCorrect = 0, flagsCategory = 0

  for (const row of picksRows ?? []) {
    const score = calculatePreSeasonPoints(
      { top4: row.top4, tenth_place: row.tenth_place, relegated: row.relegated, promoted: row.promoted, promoted_playoff_winner: row.promoted_playoff_winner },
      actuals,
    )

    // Upsert pre_season_awards — PRESERVE confirmed flag for already-confirmed rows
    const { data: existing } = await admin.from('pre_season_awards')
      .select('confirmed').eq('member_id', row.member_id).eq('season', season).maybeSingle()
    const preserveConfirmed = existing?.confirmed === true

    await admin.from('pre_season_awards').upsert({
      member_id: row.member_id,
      season,
      calculated_points: score.totalPoints,
      awarded_points: preserveConfirmed ? undefined : score.totalPoints, // don't overwrite George's adjustment
      flags: score.flags,
      ...(preserveConfirmed ? {} : { confirmed: false, confirmed_by: null, confirmed_at: null }),
    } as any, { onConflict: 'member_id,season' })

    // Emit admin_notifications — wrap in try/catch (Pattern 5)
    try {
      if (score.flags.all_correct_overall) {
        await admin.from('admin_notifications').insert({ type: 'pre_season_all_correct', metadata: { member_id: row.member_id, season } })
        flagsAllCorrect++
      } else {
        const cats = []
        if (score.flags.all_top4_correct) cats.push('top4')
        if (score.flags.all_relegated_correct) cats.push('relegated')
        if (score.flags.all_promoted_correct) cats.push('promoted')
        if (cats.length) {
          await admin.from('admin_notifications').insert({ type: 'pre_season_category_correct', metadata: { member_id: row.member_id, season, categories: cats } })
          flagsCategory++
        }
      }
    } catch { /* notification failure does not fail calc */ }
  }

  // Single "awards_ready" notification
  try {
    await admin.from('admin_notifications').insert({ type: 'pre_season_awards_ready', metadata: { season, count: picksRows?.length ?? 0 } })
  } catch { /* */ }

  revalidatePath('/admin/pre-season')
  return { success: true as const, awardsCreated: picksRows?.length ?? 0, flagsEmitted: { all_correct: flagsAllCorrect, category: flagsCategory } }
}

export async function confirmPreSeasonAward(formData: FormData) {
  const auth = await requireAdmin()
  if ('error' in auth) return auth
  const parsed = confirmPreSeasonAwardSchema.safeParse(Object.fromEntries(formData))
  if (!parsed.success) return { error: parsed.error.issues[0]?.message ?? 'Invalid input' }
  const { member_id, season, override_points } = parsed.data

  const admin = createAdminClient()
  const { data: existing } = await admin.from('pre_season_awards')
    .select('calculated_points').eq('member_id', member_id).eq('season', season).maybeSingle()
  if (!existing) return { error: 'No calculated award found — run calculation first' }

  const awarded = override_points ?? existing.calculated_points

  const { error } = await admin.from('pre_season_awards').upsert({
    member_id, season,
    awarded_points: awarded,
    confirmed: true,
    confirmed_by: auth.userId,
    confirmed_at: new Date().toISOString(),
  } as any, { onConflict: 'member_id,season' })
  if (error) return { error: error.message }

  revalidatePath('/admin/pre-season')
  return { success: true as const }
}
```

export async function bulkConfirmPreSeasonAwards(formData: FormData) {
  const auth = await requireAdmin()
  if ('error' in auth) return auth
  const season = Number(formData.get('season'))
  if (!Number.isInteger(season)) return { error: 'Invalid season' }

  const admin = createAdminClient()
  const { data: pending } = await admin.from('pre_season_awards')
    .select('member_id, calculated_points').eq('season', season).eq('confirmed', false)

  let confirmedCount = 0
  for (const row of pending ?? []) {
    const { error } = await admin.from('pre_season_awards').update({
      awarded_points: row.calculated_points,
      confirmed: true,
      confirmed_by: auth.userId,
      confirmed_at: new Date().toISOString(),
    }).eq('member_id', row.member_id).eq('season', season)
    if (!error) confirmedCount++
  }

  revalidatePath('/admin/pre-season')
  return { success: true as const, confirmedCount }
}
```
