```typescript
import { createAdminClient } from '@/lib/supabase/admin'

export interface SeasonRow {
  id: number; season: number; label: string; gw1_kickoff: string;
  final_top4: string[]; final_tenth: string | null;
  final_relegated: string[]; final_promoted: string[];
  final_playoff_winner: string | null; actuals_locked_at: string | null;
}

export async function getCurrentSeason(): Promise<SeasonRow | null> {
  const c = createAdminClient()
  const { data } = await c.from('seasons').select('*')
    .lte('gw1_kickoff', new Date().toISOString())
    .order('season', { ascending: false }).limit(1).maybeSingle()
  return (data as SeasonRow) ?? null
}

export async function getUpcomingSeason(): Promise<SeasonRow | null> {
  const c = createAdminClient()
  const { data } = await c.from('seasons').select('*')
    .gt('gw1_kickoff', new Date().toISOString())
    .order('season', { ascending: true }).limit(1).maybeSingle()
  return (data as SeasonRow) ?? null
}
```
