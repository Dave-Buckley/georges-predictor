    - SCORE_EVALUABLE set: 'Golden Glory', 'Jose Park The Bus'

    - rawTotal = basePoints + (bonusIncluded ? bonusPoints : 0)
    - displayTotal = isDoubleBubble ? rawTotal * 2 : rawTotal
